Este ejercicio los une todos. Repasa la idea central de cada uno antes de empezar.

// CLOSURE: variable privada que solo los métodos retornados pueden ver
function crearAlgo() {
  const _privado = []; // nadie fuera puede acceder aquí

  return {
    agregar(dato, callback) {
      _privado.push(dato);       // closure: accede a _privado
      if (typeof callback === "function") callback(dato); // callback
    },
    leer() {
      return [..._privado];      // closure: retorna copia con spread
    },
    total() {
      return this.leer().length; // this: llama al método del objeto
    },
  };
}
